> 소수 찾기 : [GO 42839]

### CODE
```java
class Solution {
	int answer = 0;
	ArrayList<String> list = new ArrayList<>();

	public int solution(String numbers) {
		String[] array = numbers.split("");
		for (int i = 0; i < array.length; i++)
			permutation(array, 0, array.length, i + 1);

		for (String str : list) {
			if (check(Integer.parseInt(str)) == 0)
				answer++;
			System.out.println(str);
		}
		return answer;
	}

	public int check(int number) {
		if (number == 1)
			return 1;
		for (int i = 2; i < number; i++)
			if (number % i == 0)
				return 1;
		return 0;
	}

	public void permutation(String[] arr, int depth, int n, int r) {
		String temp;
		if (depth == r) {
			temp = combine(arr, r);
			if (!temp.equals("") && !list.contains(temp))
				list.add(temp);
			return;
		}

		for (int i = depth; i < n; i++) {
			swap(arr, depth, i);
			permutation(arr, depth + 1, n, r);
			swap(arr, depth, i);
		}
	}

	public String combine(String[] arr, int r) {
		String s = "";
		String res = "";
		for (int i = 0; i < r; i++)
			s += arr[i];
		res = s.replaceAll("^0+", "");
		return res;
	}

	public void swap(String[] arr, int depth, int i) {
		String temp = arr[depth];
		arr[depth] = arr[i];
		arr[i] = temp;
	}
}
```
