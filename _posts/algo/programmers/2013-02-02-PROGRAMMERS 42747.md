```java
class Solution {
	public int solution(int[] citations) {
		int answer = 0;
		int n = citations.length;
		int count1, count2;
		Arrays.sort(citations);
		int max = citations[n - 1];

		for (int i = max; i >= 0; i--) {
			count1 = 0;
			count2 = 0;
			for (int num : citations)
				if (num > i - 1)
					count1++;

			for (int num : citations)
				if (num < i + 1)
					count2++;

			if (count1 > i - 1 && count2 < i + 1) {		
				answer = i;			
				break;
			}
		}
		return answer;
	}
}
```
