> 로또 : [GO 6603]

### CODE
```java
public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		while (true) {
			int n = sc.nextInt();
			if (n == 0)
				break;

			int[] numbers = new int[n];
			boolean[] check = new boolean[n];
			ArrayList picked = new ArrayList<Integer>();

			for (int i = 0; i < n; i++)
				numbers[i] = sc.nextInt();

			Arrays.sort(numbers);
			pick(n, picked, check, numbers);
			System.out.println();
		}
	}

	public static void pick(int n, ArrayList<Integer> picked, boolean[] check, int[] numbers) {
		if (picked.size() == 6) {
			solution(picked);
			return;
		}

		int small = picked.isEmpty() ? 0: picked.get(picked.size() - 1);
		for (int i = 0; i < n; i++) {
			if (check[i])
				continue;
			if (numbers[i] < small)
				continue;
			check[i] = true;
			picked.add(numbers[i]);
			pick(n, picked, check, numbers);
			picked.remove(picked.size() - 1);
			check[i] = false;
		}
	}

	public static void solution(ArrayList<Integer> picked) {
		for (int i = 0; i < picked.size(); i++)
			System.out.print(picked.get(i) + " ");
		System.out.println();
	}
}
```
