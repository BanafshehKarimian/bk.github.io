> 영역 구하기 : [GO 2583]

### CODE

```java
public class Main {
	public static int[] dx = { 0, 0, 1, -1 };
	public static int[] dy = { 1, -1, 0, 0 };
	public static int[][] numbers, temp;
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int m = sc.nextInt(); // 세로
		int n = sc.nextInt(); // 가로
		int k = sc.nextInt();

		numbers = new int[n][m];
		temp = new int[n][m];

		for (int j = 0; j < m; j++) // 5
			for (int i = 0; i < n; i++) // 7
				numbers[i][j] = 0;

		for (int index = 0; index < k; index++) {
			int x1 = sc.nextInt();
			int y1 = sc.nextInt();
			int x2 = sc.nextInt();
			int y2 = sc.nextInt();
			for (int j = y1; j < y2; j++)
				for (int i = x1; i < x2; i++)
					numbers[i][j] = 101;
		}

		int count = solve(numbers, temp, 0, m, n);
		System.out.println(count);
		for (int number : result(temp, count, m , n))
			System.out.print(number + " ");
	}

	public static ArrayList<Integer> result(int[][] temp, int count, int m, int n) {
		ArrayList <Integer> res = new ArrayList<>();
		for (int i = 1; i <= count; i++) {
			int result = 0;
			for (int r = 0; r < m; r++)
				for (int c = 0; c < n; c++)
					if (temp[c][r] == i)
						result++;
			res.add(result);
		}
		Collections.sort(res);
		return res;
	}

	public static int solve(int[][] numbers, int[][] temp, int count, int m, int n) {
		for (int j = 0; j < m; j++)
			for (int i = 0; i < n; i++)
				if (numbers[i][j] != 101 && temp[i][j] == 0)
					dfs(numbers, temp, i, j, ++count, m, n);
		return count;
	}

	public static void dfs(int[][] numbers, int[][] temp, int x, int y, int count, int m, int n) {
		temp[x][y] = count;
		for (int i = 0; i < 4; i++) {
			int nx = x + dx[i];
			int ny = y + dy[i];
			if (nx >= 0 && nx < n && ny >= 0 && ny < m)
				if (numbers[nx][ny] != 101 && temp[nx][ny] == 0)
					dfs(numbers, temp, nx, ny, count, m, n);
		}
	}
}
```
### COMMENT
* 인접 행렬 + DFS
* dx, dy : 상하좌우 이동을 위한 배열
* temp : 모눈종이 결과물 배열
