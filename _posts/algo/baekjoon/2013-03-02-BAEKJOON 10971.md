> 외판원 순회 2 : [GO 10971]

### CODE
```java
public class Main {
	public static int answer = Integer.MAX_VALUE;

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		ArrayList picked = new ArrayList<Integer>();
		int n = sc.nextInt();
		boolean[] check = new boolean[n];
		int[][] array = new int[n][n];
		for (int i = 0; i < n; i++)
			for (int j = 0; j < n; j++)
				array[i][j] = sc.nextInt();
		pick(n, picked, check, array);
		System.out.println(answer);
	}

	public static void pick(int n, ArrayList<Integer> picked, boolean[] check, int[][] array) {
		if (picked.size() == n) {
			solution(n, picked, array);
			return;
		}

		for (int i = 0; i < n; i++) {
			if (check[i])
				continue;
			check[i] = true;
			picked.add(i);
			pick(n, picked, check, array);
			picked.remove(picked.size() - 1);
			check[i] = false;
		}
	}

	public static void solution(int n, ArrayList<Integer> picked, int[][] array) {
		boolean tf = true;
		int sum = 0;
		for (int i = 0; i < n - 1; i++) {
			int number = array[picked.get(i)][picked.get(i + 1)];
			if (number == 0)
				tf = false;
			else
				sum += number;
		}
		if (tf && array[picked.get(n - 1)][picked.get(0)] != 0) {
			sum += array[picked.get(n - 1)][picked.get(0)];
			if (answer > sum)
				answer = sum;
		}
	}
}
```
