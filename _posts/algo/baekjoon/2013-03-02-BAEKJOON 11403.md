### CODE

```java
public class Main {
	public static int[][] numbers, result;
	public static boolean[] check;
	public static int n;

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		n = sc.nextInt();
		numbers = new int[n + 1][n + 1];
		result = new int[n + 1][n + 1];
		check = new boolean[n + 1];

		for (int i = 1; i < n + 1; i++)
			for (int j = 1; j < n + 1; j++)
				numbers[i][j] = sc.nextInt();

		solve(numbers, check);
		print(result);
	}

	public static void solve(int[][] numbers, boolean[] check) {
		for (int i = 1; i < n + 1; i++) {
			dfs(numbers, check, i);
			for (int j = 1; j < n + 1; j++) {
				if (check[j])
					result[i][j] = 1;
				else
					result[i][j] = 0;			
			}
			Arrays.fill(check, false);
		}
	}

	public static void dfs(int[][] numbers, boolean[] check, int i) {
		for (int j = 1; j < n + 1; j++) {
			if (numbers[i][j] == 1 && !check[j]){		
				check[j] = true;
				dfs(numbers, check, j);
			}
		}
	}

	public static void print(int[][] result) {
		for (int i = 1; i < n + 1; i++) {
			for (int j = 1; j < n + 1; j++)
				System.out.print(result[i][j] + " ");
			System.out.println();
		}
	}
}
```
### COMMENT
* 인접 행렬 + DFS
