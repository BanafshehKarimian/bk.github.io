> 회의실배정 [GO 1931]

### CODE
```java
class Main {
	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringBuilder sb = new StringBuilder();
		ArrayList<Work> list = new ArrayList<>();
		int n = Integer.parseInt(br.readLine());
		for (int i = 0; i < n; i++) {
			StringTokenizer st = new StringTokenizer(br.readLine());
			Work work = new Work(Integer.parseInt(st.nextToken()), Integer.parseInt(st.nextToken()));
			list.add(work);
		}
		Collections.sort(list, Work.startCompare());
		Collections.sort(list, Work.endCompare());

    int result = 1;
		int templast = list.get(0).getEnd();
		for (int i = 1; i < n; i++) {
			int first = list.get(i).getStart();
			int last = list.get(i).getEnd();
			if (first >= templast) {
				result++;
				templast = last;
			}
		}
		sb.append(result);
		System.out.println(sb);
	}
}

class Work {
	private final Integer start;
	private final Integer end;

	public int getStart() {
		return this.start;
	}

	public int getEnd() {
		return this.end;
	}

	public Work(int start, int end) {
		this.start = start;
		this.end = end;
	}

	static Comparator<Work> startCompare() {
		return new Comparator<Work>() {
			public int compare(Work x, Work y) {
				return x.start.compareTo(y.start);
			}
		};
	}

	static Comparator<Work> endCompare() {
		return new Comparator<Work>() {
			public int compare(Work x, Work y) {
				return x.end.compareTo(y.end);
			}
		};
	}
}
```
