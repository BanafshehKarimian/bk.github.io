> 음식물 피하기 : [GO 1743]

### CODE

```java
public class Main {
	public static int[] dx = { 0, 0, 1, -1 };
	public static int[] dy = { 1, -1, 0, 0 };
	public static String[][] numbers;
	public static int[][] temp;

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int m = sc.nextInt();
		int k = sc.nextInt();
		numbers = new String[m + 1][n + 1];
		temp = new int[m + 1][n + 1];
		for (int r = 1; r < n + 1; r++)
			for (int c = 1; c < m + 1; c++)
				numbers[c][r] = ".";

		for (int i = 0; i < k; i++) {
			int c = sc.nextInt();
			int r = sc.nextInt();
			numbers[r][c] = "#";
		}

		System.out.println(result(temp, solve(numbers, temp, 0, m, n), m, n));
	}

	public static int result(int[][] temp, int count, int m, int n) {
		int max = 0;
		for (int i = 1; i <= count; i++) {
			int result = 0;
			for (int r = 1; r < n + 1; r++)
				for (int c = 1; c < m + 1; c++)
					if (temp[c][r] == i)
						result++;
			if (result > max)
				max = result;
		}
		return max;
	}

	public static int solve(String[][] numbers, int[][] temp, int count, int m, int n) {
		for (int j = 1; j < n + 1; j++)
			for (int i = 1; i < m + 1; i++)
				if (numbers[i][j] == "#" && temp[i][j] == 0)
					dfs(numbers, temp, i, j, ++count, m, n);
		return count;
	}

	public static void dfs(String[][] numbers, int[][] temp, int x, int y, int count, int m, int n) {
		temp[x][y] = count;
		for (int i = 0; i < 4; i++) {
			int nx = x + dx[i];
			int ny = y + dy[i];
			if (nx >= 1 && nx < m + 1 && ny >= 1 && ny < n + 1)
				if (numbers[nx][ny] == "#" && temp[nx][ny] == 0)
					dfs(numbers, temp, nx, ny, count, m, n);
		}
	}
}
```
### COMMENT
* 인접 행렬 + DFS
* dx, dy : 상하좌우 이동을 위한 배열
* temp : 음식물 피하기 결과물 배열
