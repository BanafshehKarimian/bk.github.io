> 체스판 다시 칠하기 [GO 1018]

### CODE
```java
class Main {
	static int n, m;
	static String[][] input, result1, result2;
	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringBuilder sb = new StringBuilder();
		StringTokenizer st = new StringTokenizer(br.readLine());
		n = Integer.parseInt(st.nextToken());
		m = Integer.parseInt(st.nextToken());
		input = new String[m][n];
		result1 = new String[m][n];
		result2 = new String[m][n];
		for (int j = 0; j < n; j++) {
			for (int i = 0; i < m; i++) {
				if ((i + j) % 2 == 0) {
					result1[i][j] = "B";
					result2[i][j] = "W";
				} else {
					result2[i][j] = "B";
					result1[i][j] = "W";
				}
			}
		}

		for (int j = 0; j < n; j++) {
			String str = new StringTokenizer(br.readLine()).nextToken();
			for (int i = 0; i < m; i++)
				input[i][j] = str.charAt(i) + "";
		}

		int result = Integer.MAX_VALUE;
		for (int j = 0; j <= n - 8; j++)
			for (int i = 0; i <= m - 8; i++)
				result = Math.min(result, solve(i,j));

		sb.append(result);
		System.out.println(sb);
	}

	static int solve(int x, int y) {
		int count1 = 0
		int count2 = 0;
		for (int j = y; j < y + 8; j++) {
			for (int i = x; i < x + 8; i++) {
				if (!result1[i][j].equals(input[i][j]))
					count1++;
				if (!result2[i][j].equals(input[i][j]))
					count2++;
			}
		}
		return Math.min(count1, count2);
	}
}
```
