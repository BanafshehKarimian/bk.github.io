### CODE
```java
public class Main {
	public static int answer = 0;
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int[] numbers = new int[n];
		boolean[] check = new boolean[n];
		ArrayList picked = new ArrayList<Integer>();

		for (int i = 0; i < n; i++)
			numbers[i] = sc.nextInt();

		Arrays.sort(numbers);
		pick(n, picked, check, numbers);
		System.out.println(answer);
	}

	public static void pick(int n, ArrayList picked, boolean[] check, int[] numbers) {
		if (picked.size() == n) {
			int sum = solution(picked);
			if (sum > answer)
				answer = sum;
			return;
		}

		for (int i = 0; i < n; i++) {
			if (check[i])
				continue;
			check[i] = true;
			picked.add(numbers[i]);
			pick(n, picked, check, numbers);
			picked.remove(picked.size() - 1);
			check[i] = false;
		}
	}

	public static int solution(ArrayList<Integer> picked) {
		int sum = 0;
		for (int i = 1; i < picked.size(); i++)
			sum += Math.abs(picked.get(i - 1) - picked.get(i));
		return sum;
	}
}
```
