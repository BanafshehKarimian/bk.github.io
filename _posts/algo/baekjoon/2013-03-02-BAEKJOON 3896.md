> 소수 사이 수열 [GO 3896]

### CODE
```java
class Main {
	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		int t = Integer.parseInt(br.readLine());
		boolean[] array = new boolean[1299709 + 1];
		array[1] = true;
		for (int i = 2; i <= (int) Math.sqrt(1299709); i++) {
			if (array[i])
				continue;
			for (int j = i + i; j <= 1299709; j += i)
				array[j] = true;
		}
		for (int i = 0; i < t; i++) {
			int k = Integer.parseInt(br.readLine());
			if (k == 1)
				System.out.println(0);
			System.out.println(solve(array, k));
		}
	}

	public static int solve(boolean[] array, int m) {
		if (!array[m])
			return 0;
		else {
			int left = m;
			int right = m;
			while (true)
				if (!array[--left])
					break;
			while (true)
				if (!array[++right])
					break;
			return right - left;
		}
	}
}
```
